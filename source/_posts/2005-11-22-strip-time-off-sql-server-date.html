--- layout: post title: "Strip Time Off SQL Server Date Variable" date: 2005-11-22T13:38:00-08:00 ---
{

DECLARE @DOUT DATETIME
SET @DOUT = CONVERT(DATETIME, CONVERT(CHAR(10), GETDATE(), 101))
PRINT CONVERT(DATETIME, @DOUT)

}
